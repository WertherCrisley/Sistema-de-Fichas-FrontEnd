app.post("/fichas", async (req, res) => {
    try {
        const { tipo } = req.body;

        // Valida o tipo informado
        if (!["NORMAL", "PREFERENCIAL"].includes(tipo)) {
            return res.status(400).json({
                erro: "Tipo de ficha inválido."
            });
        }

        // Busca a última ficha criada
        const ultimaFicha = await prisma.ficha.findFirst({
            orderBy: {
                criadaEm: "desc"
            }
        });

        // Gera o próximo código
        let codigo = "A1";

        if (ultimaFicha) {
            const numero = parseInt(ultimaFicha.codigo.slice(1));
            codigo = `A${numero + 1}`;
        }

        // Cria a ficha
        const ficha = await prisma.ficha.create({
            data: {
                codigo,
                tipo
            }
        });

        return res.status(201).json(ficha);

    } catch (erro) {
        console.error(erro);

        return res.status(500).json({
            erro: "Erro ao criar ficha."
        });
    }
});







model Ficha {
  id           String       @id @default(auto()) @map("_id") @db.ObjectId
  codigo       String       @unique
  tipo         TipoFicha
  status       StatusFicha  @default(AGUARDANDO)
  criadaEm     DateTime     @default(now())
  chamadaEm    DateTime?
  finalizadaEm DateTime?
}

enum TipoFicha {
  NORMAL
  PREFERENCIAL
}

enum StatusFicha {
  AGUARDANDO
  EM_ATENDIMENTO
  FINALIZADA
}
